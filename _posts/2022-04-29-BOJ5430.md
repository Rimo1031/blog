## 코드

- 사용 언어 : C++

```cpp
#include <iostream>
#include <deque>
#include <string>
#include <sstream>

using namespace std;

string AC(deque<int> s, string func) {
    string res;
    string error = "error\n";
    int isReversed = 0;
    for (int i=0;i<func.size();i++) {
        if (func[i] == 'R') {
            isReversed = (isReversed+1) % 2;
        }
        if (func[i] == 'D') {
            if (s.empty()) return error;
            else {
                if (isReversed) {
                    s.pop_back();
                }
                else {
                    s.pop_front();
                }
            }
        }
    }
    if (s.empty()) {
        res = "[]\n";
    }
    else {
        res.append("[");
        while (s.size() > 1) {
            if (isReversed) {
                res.append(to_string(s.back()) + ",");
                s.pop_back();
            }
            else {
                res.append(to_string(s.front()) + ",");
                s.pop_front();
            }
        }
        res.append(to_string(s.back()) + "]\n");
    }
    return res;
}

int main() {
	int tc;
	cin >> tc;
	for (int i=0;i<tc;i++) {
		string func;
		int n;
		string input;
		cin >> func;
		cin >> n;
		cin >> input;
		input = input.substr(1, input.size()-2);
		istringstream ss;
		ss.str(input);
		deque<int> s;
		for (string num;getline(ss, num, ','); ) {
			s.push_back(stoi(num));
		}
		cout << AC(s, func);
	}
	return 0;
}
```
